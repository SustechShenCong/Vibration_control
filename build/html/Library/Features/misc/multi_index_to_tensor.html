function subs = multi_index_to_sub(p)
    d = sum(p);
    subs = zeros(1,d);
    [~, i, s] = find(p);
    m = 1;
    for k = 1:d
        subs(k) = i(m);
        if s(m) == 1
            m = m + 1;
        else
            s(m) = s(m) - 1;
        end
    end
end
